RewriteEngine On

# If the request is for a real file, serve it directly
RewriteCond %{REQUEST_FILENAME} -f
RewriteRule ^ - [L]

# If the request is for a real directory, serve it directly
RewriteCond %{REQUEST_FILENAME} -d
RewriteRule ^ - [L]

# Otherwise, pass everything to Node.js
RewriteRule ^(.*)$ http://127.0.0.1:3000/$1 [L,P]
